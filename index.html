resourced (express.js)

A REST/resource-oriented DSL for configuring express.js which makes it easy to defined resources, include links, perform validation.

Usage

The following a simple resource which responds to GET requests:

module.exports = new Resource({
    url: "/people/:id",
    urlSchema: {
        id: mustBe().numeric().populated()
    },

    cache: caching.minutes(5).publically(),

    respondsTo: [
        http.get(function(id) {
            associatedAddressLink = addressResource.getLink("address", { id: "5"});

            return {
                firstName : "Colin Jack",
                address: associatedAddressLink
            };
        })
    ]
});
